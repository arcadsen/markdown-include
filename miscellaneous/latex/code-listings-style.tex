# 这是一个Python示例
def fibonacci(n):
    """计算斐波那契数列"""
    a, b = 0, 1
    for _ in range(n):
        print(a, end=' ')
        a, b = b, a + b
    print()

fibonacci(10)  # 输出前10个斐波那契数
