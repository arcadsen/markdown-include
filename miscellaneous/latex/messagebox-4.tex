\usepackage{lipsum,lmodern}
\usepackage[most]{tcolorbox}

\begin{document}

\lipsum[1]

  enhanced jigsaw,
  breakable,
  pad at break*=1mm,
  colback=green!25!yellow!10!white,
  colframe=green!75!black,
  title=渲染结果,
  drop fuzzy shadow,]
  \lipsum[1-12]
\end{tcolorbox}
